Task 1: Rotate the stepper motor 1/12 rotation (use full step)
Task 2: move the servo motor arm to 0 degree for 1 second, then move it to 120 for 3 seconds then move it back to 0.
Task 3: Rotate the servo motor arm gradullay from 0 to 180 (4 seconds) and then back to 0.



#include <xc.h>
#include <pic18f4550.h> 

#pragma config PBADEN	= OFF 
#pragma config FOSC 	= HS
#pragma config PWRT 	= OFF
#pragma config BOR 		= OFF
#pragma config ICPRT 	= OFF
#pragma config LVP 		= OFF
#pragma config WDT 		= OFF
#pragma config DEBUG	= OFF
#pragma config CPUDIV	= OSC1_PLL2	


#define _XTAL_FREQ 8000000  


void main(void)
{
    
    TRISB = 0;
    
    unsigned int i = 2048 / 4;
    i = i / 12;
    while (i--)
    {
        PORTB = 0b00001000;
        __delay_ms(20);
        PORTB = 0b00010000;
        __delay_ms(20);
        PORTB = 0b00100000;
        __delay_ms(20);
        PORTB = 0b01000000;
        __delay_ms(20);
    }
    while (1);
    return;
}
